自定义图片懒加载指令

2022-07-12 Vue 176 words
vue3

定义指令

import defaultImg from '@/assets/images/200.png'

export default {
  install(app) {
    // 定义指令
    defineDerective(app)
  }
}
// 定义指令
const defineDerective = (app) => {
  app.directive('lazy', {
    // el是使用指令的元素，binding是给指令传递的参数
    mounted(el, binding) {
      // 创建一个观察对象，用于观察元素是否进入可视区域
      const observe = new IntersectionObserver(
        ([{ isIntersecting }]) => {
          if (isIntersecting) {
            // 停止观察
            observe.unobserve(el)
            // 处理图片加载失败
            el.onerror = () => {
              // 加载失败设置默认图
              el.src = defaultImg
            }
            // 把指令的值设置给el的src属性
            el.src = binding.value
          }
        },
        {
          // 元素与可视区域相交的阈值
          threshold: 0
        }
      )
      // 开启观察
      observe.observe(el)
    }
  })
}

使用
