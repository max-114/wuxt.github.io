express模板

2022-04-07 Nodejs 124 words
express 模板 
const express = require('express')
const app = express()

// 中间件的使用
app.use(express.urlencoded({ extended: false }))

// 自定义中间件
app.use((req, res, next) => {
  res.cc = function (msg, status = 1) {
    res.send({
      status,
      message: msg
    })
  }
  next()
})

app.get('/', (req, res) => {
  res.send({
    status: 200,
    msg: 'hi'
  })
})

app.post('/', (req, res) => {
  res.cc('hi', 200)
})

// 错误处理中间件（放在最后才能捕获错误）
app.use((err, req, res, next) => {
  res.send({
    status: 500,
    msg: err.message
  })
})

app.listen(8080, () => {
  console.log('server running at http://127.0.0.1:8080')
})
