socket.io笔记

2022-04-18 第三方库 697 words

基本使用

server

安装

npm i socket.io

使用

const io = require('scoket.io')(3000，{
  cors:{
    // 数组中传入允许跨域请求的客户端地址，也可以直接赋值为'*'，允许所有跨域请求
    origin:['http://localhost:8080']
  }
}) // 3000为服务的端口号

io.on('connection', socket => {
  console.log('a user connected')

  socket.on('message',message=>{
    // 向所有客户端发送消息事件，包括发送消息的客户端
    io.emit('message',message)
    // 向所有客户端发送消息事件，除了发送消息的客户端
    socket.broadcast.emit('message',message)
  })
})

client

安装

npm i socket.io-client
import { io } from 'socket.io-client'

const socket = io(url) // url为服务器的url地址

socket.emit('message', message)

socket.on('message', message => {
  console.log(message)
})

常用 API

emit 的最后一个参数

该参数是一个回调函数,让服务端可以调用到客户端的函数
server

io.on('connection', socket => {
  socket.on('join room', (room, callback) => {
    socket.join(room)
    callback(`joined ${room}`)
  })
})

client

const displayMessag = () => {}
...
socket.emit('join room', room, message => {
  displayMessag(message)
})

to()

向指定的 room 发送消息事件,只有 room 中的成员能够监听到消息事件
案例:向某个指定用户发送消息

socket.io 中每个用户的 socket.id 默认是他们当前所在的 room

server

io.on('connection', socket => {
  socket.on('message', (message, room) => {
    // to 本身具备broadcast的功能
    socket.to(room).emit('message', message)
  })
})

client

// room为消息发送对象的socket.id
socket.emit('message', message, room)

join()

将用户加入 room 中,每个用户可以加入多个 room
server

io.on('connection', socket => {
  socket.on('join room', room => {
    socket.join(room)
  })
})

client

// room为消息发送对象的socket.id
socket.emit('join room', room)

Namespace

server

const io = require('scoket.io')(3000，{
  cors:{
    origin:['http://localhost:8080']
  }
})

const userIo = io.of('/user)
// 创建namespace
userIo.on('connection', socket => {
  console.log('connected to user namespace)
})

client

import { io } from 'socket.io-client'
// 连接namespace
const userSocket = io('http://localhost:3000/user')

中间件 Middleware

server

const io = require('scoket.io')(3000，{
  cors:{
    origin:['http://localhost:8080']
  }
})

const userIo = io.of('/user)

userIo.use((socket,next)=>{
  if(socket.handshake.auth.token){
    socket.username = getUsernameFromToken(socket.handshake.auth.token)
    next()
  }else{
    // 客户端可以通过监听connect_error事件捕获该错误
    next(new Error('please send token'))
  }
})

userIo.on('connection', socket => {
  console.log('connected to user namespace with username' + socket.username)
})

function getUsernameFromToken(token){
  ...
  return username
}

client

import { io } from 'socket.io-client'
// 连接时传入token
const userSocket = io('http://localhost:3000/user', { auth: { token: 'test' } })

userSocket.on('connect_error', error => {
  console.log(error)
})

admin-ui

server

npm i @socket.io/admin-ui
const { createServer } = require('http')
const { Server } = require('socket.io')
const { instrument } = require('@socket.io/admin-ui')

const httpServer = createServer()

const io = new Server(httpServer, {
  // 配置跨域
  cors: {
    origin: ['https://admin.socket.io'],
    credentials: true
  }
})

instrument(io, {
  auth: false
})

httpServer.listen(3000)

安装完成后需要重启服务端

在浏览器打开https://admin.socket.io/

输入服务端地址+/admin(例如:http://localhost:3000/admin)
