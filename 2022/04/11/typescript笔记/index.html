typescript笔记

2022-04-11 Typescript 1.4k words

常用命令

生成配置文件

tsc --init
# 生成tsconfig.json

编译为 js 文件

tsc xxx.ts

自动编译

# ts文件变动时自动编译
tsc --watch

发出错误

# ts文件报错时不编译为js
tsc -noEmitOnError hello .ts

配置

配置文件：tsconfig.json

ES 降级编译

/* 编译为符合es5规范的js文件 */
"target":"es5"

严格模式

/* 默认开启所有严格检查 */
"strict": true,
/* 不允许存在隐式any类型 */
"noImplicitAny": true,
/* 不允许存在null值 */
"strictNullChecks": true

语法

type 和 interface 区别

类型扩展

type

type Animal = {
  name: string
}
type Bear = Animal & {
  honey: boolean
}

interface

interface Animal {
  name: string
}
interface Bear extends Animal {
  honey: boolean
}
const bear: Bear = {
  name: 'winie',
  honey: true
}

向现有类型添加字段

interface

// 可以通过定义同名的interface添加字段
interface myWindow {
  count: number
}
interface myWindow {
  title: string
}
const w: myWindow = {
  title: 'hello',
  count: 100
}

type

type 一旦定义，不能改变

type myWindow = {
  count: number
}
type myWindow = {
  title: string
}
// Error: Duplicate identifier 'Window'.

类型断言

as

const myCanvas = document.getElementById('main_canvas') as HTMLCanvasElement

****前缀

const myCanvas = <HTMLCanvasElement>document.getElementById("main_canvas")

！后缀：表示从该值的类型中去除了null和undefined

const myCanvas = document.getElementById('main_canvas')!

可选参数

**参数名?**：表示该参数为可选参数

interface Animal {
  name: string
  // hobby的类型为 string|undefined
  hobby?: string
}

文字类型

function handleRequest(method: 'get' | 'post') {}
// ok
const method = 'get'
handleRequest(method)
const req = {
  method: 'get'
}
// 类型“string”的参数不能赋给类型“"get" | "post"”的参数
handleRequest(req.method)

可以通过以下方式解决

类型断言

handleRequest(req.method as 'get')
handleRequest(<'get'>req.method)
const req = {
  method: 'get' as 'get'
}

as const后缀：表示所有属性类型均为赋值时的文字类型，而非更宽泛的string或number等

const req = {
  method: 'get'
} as const

类型缩小

等值缩小

// 当x和y相等时，typescript推断x和y均为string类型
function example(x: string | number, y: string | boolean) {
  if (x === y) {
    y.toUpperCase()
    x.toUpperCase()
  }
}
// typescript推断x为number和string的联合类型
let x = Math.random() < 0.5 ? 10 : 'hello'
x = 1
console.log(x)
x = 'good'
console.log(x)
// 1
// good

类型谓词

格式：parameterName is Type

// 自定义类型守卫isFish
function isFish(pet: Fish | Bird): pet is Fish {
  return (pet as Fish).swim !== undefined
}
let pet = getSmallPet() // Fish or Bird
if (isFish(pet)) pet.swim()
else pet.fly()
const zoo: (Fish | Bird)[] = [shark, sparrow, whale]
const underWater: Fish[] = zoo.filter((pet): pet is Fish => {
  if (pet.name === 'frog') return false
  return isFish(pet)
})

函数

调用签名 Call Signatures

定义具有额外属性的函数类型

type DescribableFunction = {
  description: string
  (someArg: number): boolean
}
function doSomething(fn: DescribableFunction) {
  console.log(fn.description + ' returned ' + fn(6))
}

function fn(num: number): boolean {
  return num ? true : false
}
fn.description = 'fn'
doSomething(fn)
// fn returned true

构造签名 Construct Signatures

定义构造函数的类型

type SomeConstructor = {
  new (s: string): Ctor
}
class Ctor {
  s: string
  constructor(s: string) {
    this.s = s
  }
}
function fn(ctor: SomeConstructor) {
  return new ctor('hello')
}
console.log(fn(Ctor).s) // hello

泛型函数 Generic Functions

可以将函数参数类型和返回值类型关联起来

function firstElement<Type>(arr: Type[]): Type | undefined {
  return arr[0]
}
console.log(firstElement([1]))

可以同时使用多个类型参数

function map<Input, Output>(
  arr: Input[],
  func: (arg: Input) => Output
): Output[] {
  return arr.map(func)
}

// Parameter 'n' is of type 'string'
// 'parsed' is of type 'number[]'
const parsed = map(['1', '2', '3'], n => parseInt(n))

In TypeScript, generics are used when we want to describe a correspondence between two values

约束 Constraints

可以通过添加约束条件，限制 Type 的类型

function longest<Type extends { length: number }>(a: Type, b: Type) {
  if (a.length >= b.length) {
    return a
  } else {
    return b
  }
}

函数重载 Function Overloads

可以通过书写多个函数签名，定义函数的多种调用方式

// 函数签名
function makeDate(timestamp: number): Date
function makeDate(m: number, d: number, y: number): Date
// 函数实现
function makeDate(mOrTimestamp: number, d?: number, y?: number): Date {
  if (d !== undefined && y !== undefined) {
    return new Date(y, mOrTimestamp, d)
  } else {
    return new Date(mOrTimestamp)
  }
}
const d1 = makeDate(12345678)
const d2 = makeDate(5, 5, 5)
const d3 = makeDate(1, 3)

返回值类型

void
通过类型别名定义的函数返回值为 void 时，实现时可以返回任何其他的值，但会被忽略

type voidFunc = () => void

const f1: voidFunc = () => {
  return true
}

当返回值被赋值给另一个变量，该变量会获得 void 类型

const v1: void = f1()

通过字面量方式定义的函数返回值类型为 void 时，该函数不能返回任何值

function f2(): void {
  // error: Type 'boolean' is not assignable to type 'void'.
  return true
}

unknown
与 any 类型类似，但不能对 unknown 类型的值进行任何操作

function f2(a: unknown) {
  // error: Object is of type 'unknown'.
  a.b()
}

never
Some functions never return a value:

function fail(msg: string): never {
  throw new Error(msg)
}
function fn(x: string | number) {
  if (typeof x === 'string') {
    // do something
  } else if (typeof x === 'number') {
    // do something else
  } else {
    x // has type 'never'!
  }
}

参数展开运算符

形参展开

function multiply(n: number, ...m: number[]) {
  return m.map(x => n * x)
}
// 'a' gets value [10, 20, 30, 40]
const a = multiply(10, 1, 2, 3, 4)

实参展开

const args = [8, 5]
// error: A spread argument must either have a tuple type or be passed to a rest parameter.
const angle = Math.atan2(...args)

利用const解决

// Inferred as 2-length tuple
const args = [8, 5] as const
// OK
const angle = Math.atan2(...args)

参数结构 Parameter Destructuring

function sum({ a, b, c }: { a: number; b: number; c: number }) {
  console.log(a + b + c)
}
// Same as prior example
type ABC = { a: number; b: number; c: number };
function sum({ a, b, c }: ABC) {
  console.log(a + b + c);
}```

对象类型

readonly属性：通过在参数名前加readonly创建一个只读的参数类型

interface SomeType {
  readonly prop: string
}

对象类型参数可以改变内部属性的值

interface Home {
  readonly resident: { name: string; age: number }
}
function evict(home: Home) {
  home.resident.name = 'name'
}
